create table TEAM(
name varchar(50) primary key,
city varchar(50) not null,
coach varchar(50) not null,
PID int not null
);

create table PLAYER(
playerID int primary key,
name varchar(50) not null,
position varchar(10) not null,
skill int not null,
tname varchar(50) references TEAM(name),
from_date date
);

alter table TEAM add team_fk foreign key(PID) references PLAYER(playerID);

create table INJURY_RECORDS(
inj_date date not null,
severity_level int not null,
description varchar(50),
PID int,
primary key(description,PID),
foreign key(PID) references PLAYER(playerID)
);

create table GAME(
host varchar(50) primary key,
guest varchar(50) primary key,
game_date date not null,
score int not null,
foreign key(host) references TEAM(name),
foreign key(host) references TEAM(name),
);
